# [1차] 뉴스 클러스터링

https://school.programmers.co.kr/learn/courses/30/lessons/17677

합집합 교집합 구현

```js
function solution(str1, str2) {
    var answer = 0;
    
    const reg=/^[a-z]*$/;
    
    const replaced_str1 = str1.toLowerCase()
    const replaced_str2 = str2.toLowerCase();
    
    let setStr1 = new Map();
    let str1_len = 0;
    let setStr2 = new Map();
    let str2_len = 0;
    
    replaced_str1.split('').map((item, index) => {
        const word = item + replaced_str1.split('')[index+1];
        
        if (index === replaced_str1.split('').length - 1) return;
        if ((word).match(reg)){
            setStr1.set(word, (setStr1.get(word) || 0) + 1 )   
            str1_len++;
        }
        return;
    })

    
    replaced_str2.split('').map((item, index) => {
        const word = item + replaced_str2.split('')[index+1];
        
        if (index === replaced_str2.split('').length - 1) return;
        if ((word).match(reg)){
            setStr2.set(word, (setStr2.get(word) || 0) + 1 ) 
            str2_len++;
        }
        return;
    })

    let 교집합_수 = 0;
    
    [...setStr1.keys()].map((item) => {
        if ([...setStr1.keys()].includes(item) && [...setStr2.keys()].includes(item)) {
            교집합_수 += setStr1.get(item) > setStr2.get(item) ? setStr2.get(item) : setStr1.get(item);
        }
    })
    
    const 합집합_수 = str1_len + str2_len - 교집합_수;
    
    if (합집합_수 === 0) return 65536;
    
    return Math.floor(교집합_수 / 합집합_수 * 65536);
}
```
