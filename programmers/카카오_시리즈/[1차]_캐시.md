function solution(cacheSize, cities) {
    let runningTime = 0;
    
    cities = cities.map(item => item.toLowerCase())
    
    const cache = [];
    
    if (cacheSize === 0 ) return 5 * cities.length;
    
    cities.map((city)=>{
        if (cache.includes(city)) {
            runningTime += 1;
            cache.splice(cache.indexOf(city),1)
            cache.push(city);
            
        } else {
            if (cache.length < cacheSize) {
            cache.push(city)
            runningTime += 5;
            } else {
            cache.shift();
            cache.push(city)
            runningTime += 5;
            }
        }
    })
    
    
    return runningTime;
}
