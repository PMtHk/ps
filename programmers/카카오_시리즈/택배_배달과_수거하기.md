# 택배 배달과 수거하기

https://school.programmers.co.kr/learn/courses/30/lessons/150369

- 스택을 활용한 구현

```js
function solution(cap, n, deliveries, pickups) {
    let answer = 0
    
    while (deliveries.length > 0 || pickups.length > 0) {
        while (deliveries.length > 0 && deliveries[deliveries.length - 1] === 0) {
            deliveries.pop()
        }
        while (pickups.length > 0 && pickups[pickups.length - 1] === 0) {
            pickups.pop()
        }
        const distance = Math.max(deliveries.length, pickups.length)
        answer += distance * 2

        let currentCap = cap;
        while (deliveries.length > 0 && currentCap > 0) {
            const lastIdx = deliveries.length - 1
            
            if (deliveries[lastIdx] > currentCap) {
                deliveries[lastIdx] -= currentCap
                currentCap = 0
            } else {
                currentCap -= deliveries[lastIdx]
                deliveries.pop()
            }
        }
        
        currentCap = cap;
        while (pickups.length > 0 && currentCap > 0) {
            const lastIdx = pickups.length - 1
            
            if (pickups[lastIdx] > currentCap) {
                pickups[lastIdx] -= currentCap
                currentCap = 0
            } else {
                currentCap -= pickups[lastIdx]
                pickups.pop()
            }
        }
    }
    
    return answer
}
```
