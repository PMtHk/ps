문자열 파싱

```js
function solution(s) {
    const answer = []
    const firstParsing = 괄호벗기기(s)
    const secondParsing = firstParsing.map(item => item.split(','))
    
    secondParsing.sort((a,b) => {
        return a.length > b.length ? 1 : -1
    })
    
    secondParsing.forEach(item => {
        for (let i=0; i<item.length; i++) {
            if (!answer.includes(parseInt(item[i]))) {
                answer.push(parseInt(item[i]))
            }
        }
    })
    
    return answer
}

function 괄호벗기기(s) {
    return s.slice(2, s.length - 2).split('},{')
}
```
